118 if (optionalPelicula.isPresent()) {

120 // Eliminar la relación en listaMeGustaPeliculas de los perfiles
121 List<Perfil> perfiles = perfilRepository.findAll();
122 for (Perfil perfil : perfiles) {
123 if (perfil.getListaMeGustaPeliculas().contains(pelicula)) {
124 perfil.getListaMeGustaPeliculas().remove(pelicula);
125 perfilRepository.save(perfil);
126 }
127 }
128 peliculaRepository.delete(pelicula);
129 } else {
130 throw new RuntimeException("Pelicula not found with id: " + id);

210 Usuario usuario = optionalUsuario.get();
211 Pelicula pelicula = optionalPelicula.get();
212 Perfil perfil = session.getAttribute(PER_STRING) != null ? (Perfil) session.getAttribute(PER_STRING)
213 : usuario.getPerfiles().get(0);
214
215 if (perfil.getListaMeGustaPeliculas().contains(optionalPelicula.get())) {
216 // Si la película ya está en la lista de favoritos, eliminarla
217 perfil.getListaMeGustaPeliculas().remove(pelicula);
218 } else {
219 // Añadir la película a la lista de favoritos del usuario
220 perfil.getListaMeGustaPeliculas().add(pelicula);
