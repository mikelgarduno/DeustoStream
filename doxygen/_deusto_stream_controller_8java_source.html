204 @ApiResponses({
205 @ApiResponse(responseCode = "201", description = "Serie creada"),
206 @ApiResponse(responseCode = "400", description = "Error al crear serie")
207 })
208 @PostMapping("/series")
209 public ResponseEntity<Series> createSeries(@RequestBody Series series) {
210 try {
211 // Crear capítulos si el número de capítulos es mayor a 0
212 if (series.getNumeroCapitulos() > 0) {
213 List<Capitulo> capitulos = new ArrayList<>();
214 for (int i = 0; i < series.getNumeroCapitulos(); i++) {
215 Capitulo capitulo = new Capitulo();
216 capitulo.setTitulo("Capítulo " + (i + 1) + " de " + series.getTitulo());
217 capitulo.setDuracion(30); // Duración por defecto
218 capitulo.setSerie(series);
219 capitulos.add(capitulo);
220 }
221 series.setCapitulos(capitulos);
222 }
223
224 Series createdSeries = deustoStreamService.createSeries(series);
225
226 return ResponseEntity.status(HttpStatus.CREATED).body(createdSeries);
227
228 } catch (Exception e) {
229 return ResponseEntity.badRequest().build();

335 if (optionalSerie.isPresent()) {
336 Series serie = optionalSerie.get();
337 capitulo.setSerie(serie);

339 int numero = (serie.getCapitulos() == null) ? 1 : serie.getCapitulos().size() + 1;
340 capitulo.setTitulo("Capítulo " + numero + " de " + serie.getTitulo());
341 }
342 if (capitulo.getDuracion() == 0) {
343 capitulo.setDuracion(30);
344 }
345 Capitulo nuevo = deustoStreamService.createCapitulo(capitulo);
