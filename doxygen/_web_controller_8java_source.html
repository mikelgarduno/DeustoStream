103 @PostMapping("/registro")
104 public String procesarRegistro(@ModelAttribute Usuario nuevoUsuario, HttpSession session, Model model) {
105 if (authService.register(nuevoUsuario)) {
106 // Buscar al usuario registrado
107 Optional<Usuario> usuarioOpt = authService.login(nuevoUsuario.getCorreo(), nuevoUsuario.getContrasenya());
108
109 if (usuarioOpt.isPresent()) {
110 Usuario usuario = usuarioOpt.get();
111 // Guardarlo en la sesión
112 session.setAttribute(USUARIO_STRING, usuario);
113
114 // Redireccionar según tipo
115 return "redirect:" + authService.obtenerRedireccion(usuario);
116 }
117 }
118
119 // Si no pudo registrarse (correo ya usado)
120 model.addAttribute("error", "El correo ya está registrado.");
121 return "registro";
122 }

187 @GetMapping("/catalogo")
188 public String catalogo(
189 @RequestParam(value = "titulo", required = false) String titulo,
190 @RequestParam(value = "genero", required = false) Generos genero,
191 @RequestParam(value = "duracion", required = false) String duracion,
192 @RequestParam(value = "anio", required = false) String anio,
193 Model model, HttpSession session) {
194 List<Pelicula> peliculas = deustoStreamService.buscarPeliculasFiltradas(titulo, genero);
195 List<Series> series = deustoStreamService.buscarSeriesFiltradas(titulo, genero);
196
197 // Filtrar por duración y año si se proporcionan
198 if (duracion != null && !duracion.isEmpty()) {
199 for (int i = 0; i < peliculas.size(); i++) {
200 int duracionPelicula = peliculas.get(i).getDuracion();
201 if (duracionPelicula > Integer.parseInt(duracion)) {
202 peliculas.remove(i);
203 i--;
204 }
205 }
206 }
207
208 if (anio != null && !anio.isEmpty()) {
209 for (int i = 0; i < peliculas.size(); i++) {
210 int anioPelicula = peliculas.get(i).getAnio();
211 if (Integer.parseInt(anio) == 2000) {
212 if (anioPelicula <= Integer.parseInt(anio)) {
213 peliculas.remove(i);
214 i--;
215 }
216 } else {
217 if (anioPelicula > Integer.parseInt(anio)) {
218 peliculas.remove(i);
219 i--;
220 }
221 }
222
223 }
224
225 for (int i = 0; i < series.size(); i++) {
226 int anioSerie = series.get(i).getAnio();
227 if (Integer.parseInt(anio) == 2000) {
228 if (anioSerie <= Integer.parseInt(anio)) {
229 series.remove(i);
230 i--;
231 }
232 } else {
233 if (anioSerie > Integer.parseInt(anio)) {
234 series.remove(i);
235 i--;
236 }
237 }
238 }
239 }
240
241 Usuario usuario = (Usuario) session.getAttribute(USUARIO_STRING);
242 Perfil perfil = session.getAttribute(PERFIL_STRING) != null ? (Perfil) session.getAttribute(PERFIL_STRING)
243 : usuario.getPerfiles().get(0);
244 model.addAttribute(PELICULA_STRING, peliculas);
245 model.addAttribute(SERIE_STRING, series);
246 model.addAttribute(PEL_FAV_STRING, perfil.getListaMeGustaPeliculas());
247 model.addAttribute(SERIES_FAV_STRING, perfil.getListaMeGustaSeries());
248 model.addAttribute(USUARIO_STRING, usuario);
249 model.addAttribute(GENEROS_STRING, Generos.values()); // Importa tu enum Generos
250 model.addAttribute("avatar", perfil.getAvatar());
251
252 return "catalogo"; // Asegúrate de que este es el nombre del archivo HTML en templates
253 }
