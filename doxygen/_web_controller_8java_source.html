96 cPass.setHttpOnly(true); // 🔒 Impide acceso vía JavaScript
97 response.addCookie(cPass);

100 cGuardar.setMaxAge(7 * 24 * 60 * 60);
101 cGuardar.setPath("/");
102 cGuardar.setSecure(true); // Asegúrate de que la cookie sea segura
103 cGuardar.setHttpOnly(true); // Impide acceso vía JavaScript
104 response.addCookie(cGuardar);
105
106 } else {
107 // Borrar cookies
108 Cookie borrarCorreo = new Cookie("correo", "");
109 borrarCorreo.setMaxAge(0);
110 borrarCorreo.setPath("/");
111 borrarCorreo.setSecure(true); // Asegúrate de que la cookie sea segura
112 borrarCorreo.setHttpOnly(true); // Impide acceso vía JavaScript
113 response.addCookie(borrarCorreo);
114

119 borrarPass.setHttpOnly(true); // Impide acceso vía JavaScript

164 @PostMapping("/registro")
165 public String procesarRegistro(@ModelAttribute Usuario nuevoUsuario, HttpSession session, Model model) {
166 if (authService.register(nuevoUsuario)) {
167 // Buscar al usuario registrado
168 Optional<Usuario> usuarioOpt = authService.login(nuevoUsuario.getCorreo(), nuevoUsuario.getContrasenya());
169
170 if (usuarioOpt.isPresent()) {
171 Usuario usuario = usuarioOpt.get();
172 // Guardarlo en la sesión
173 session.setAttribute(USUARIO_STRING, usuario);
174
175 // Redireccionar según tipo
176 return "redirect:" + authService.obtenerRedireccion(usuario);
177 }
178 }
179
180 // Si no pudo registrarse (correo ya usado)
181 model.addAttribute(ERROR_STRING, "El correo ya está registrado.");
182 return "registro";
183 }

248 @GetMapping("/catalogo")
249 public String catalogo(
250 @RequestParam(value = "titulo", required = false) String titulo,
251 @RequestParam(value = "genero", required = false) Generos genero,
252 @RequestParam(value = "duracion", required = false) String duracion,
253 @RequestParam(value = "anio", required = false) String anio,
254 Model model, HttpSession session) {
255 List<Pelicula> peliculas = deustoStreamService.buscarPeliculasFiltradas(titulo, genero);
256 List<Series> series = deustoStreamService.buscarSeriesFiltradas(titulo, genero);
257
258 // Filtrar por duración y año si se proporcionan
259 if (duracion != null && !duracion.isEmpty()) {
260 for (int i = 0; i < peliculas.size(); i++) {
261 int duracionPelicula = peliculas.get(i).getDuracion();
262 if (duracionPelicula > Integer.parseInt(duracion)) {
263 peliculas.remove(i);
264 i--;
265 }
266 }
267 }
268
269 if (anio != null && !anio.isEmpty()) {
270 for (int i = 0; i < peliculas.size(); i++) {
271 int anioPelicula = peliculas.get(i).getAnio();
272 if (Integer.parseInt(anio) == 2000) {
273 if (anioPelicula <= Integer.parseInt(anio)) {
274 peliculas.remove(i);
275 i--;
276 }
277 } else {
278 if (anioPelicula > Integer.parseInt(anio)) {
279 peliculas.remove(i);
280 i--;
281 }
282 }
283
284 }
285
286 for (int i = 0; i < series.size(); i++) {
287 int anioSerie = series.get(i).getAnio();
288 if (Integer.parseInt(anio) == 2000) {
289 if (anioSerie <= Integer.parseInt(anio)) {
290 series.remove(i);
291 i--;
292 }
293 } else {
294 if (anioSerie > Integer.parseInt(anio)) {
295 series.remove(i);
296 i--;
297 }
298 }
299 }
300 }
301
302 Usuario usuario = (Usuario) session.getAttribute(USUARIO_STRING);
303 Perfil perfil = session.getAttribute(PERFIL_STRING) != null ? (Perfil) session.getAttribute(PERFIL_STRING)
304 : usuario.getPerfiles().get(0);
305 model.addAttribute(PELICULA_STRING, peliculas);
306 model.addAttribute(SERIE_STRING, series);
307 model.addAttribute(PEL_FAV_STRING, perfil.getListaMeGustaPeliculas());
308 model.addAttribute(SERIES_FAV_STRING, perfil.getListaMeGustaSeries());
309 model.addAttribute(USUARIO_STRING, usuario);
310 model.addAttribute(GENEROS_STRING, Generos.values()); // Importa tu enum Generos
311 model.addAttribute("avatar", perfil.getAvatar());
312
313 return "catalogo"; // Asegúrate de que este es el nombre del archivo HTML en templates
314 }
