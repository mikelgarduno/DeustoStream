112 if (optionalPelicula.isPresent()) {
113 Pelicula pelicula = optionalPelicula.get();
114 // Eliminar la relación en listaMeGustaPeliculas de los perfiles
115 List<Perfil> perfiles = perfilRepository.findAll();
116 for (Perfil perfil : perfiles) {
117 if (perfil.getListaMeGustaPeliculas().contains(pelicula)) {
118 perfil.getListaMeGustaPeliculas().remove(pelicula);
119 perfilRepository.save(perfil);

122 peliculaRepository.delete(pelicula);

205 Pelicula pelicula = optionalPelicula.get();

208 if (perfil.getListaMeGustaPeliculas().contains(optionalPelicula.get())) {
209 // Si la película ya está en la lista de favoritos, eliminarla

212 // Añadir la película a la lista de favoritos del usuario
213 perfil.getListaMeGustaPeliculas().add(pelicula);
